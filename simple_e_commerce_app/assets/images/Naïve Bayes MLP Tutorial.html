import pandas as pd
from sklearn import preprocessing
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.naive_bayes import GaussianNB
from sklearn.metrics import confusion_matrix, accuracy_score, precision_score, recall_score

# Load the dataset
df = pd.read_csv('Social_Network_Ads - Social_Network_Ads.csv')  # Reading CSV file

# Check and encode categorical column 'Gender'
lab_encoder = preprocessing.LabelEncoder()
df['Gender'] = lab_encoder.fit_transform(df['Gender'])  # Male = 1, Female = 0

# Define feature matrix (Age, EstimatedSalary, Gender) and target (Purchased)
x = df[['Gender', 'Age', 'EstimatedSalary']].values  # Input features
y = df['Purchased'].values  # Target variable

# Split into training and testing sets (80% train, 20% test)
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=0)

# Feature scaling using Standard Scaler
sc = StandardScaler()
x_train = sc.fit_transform(x_train)
x_test = sc.transform(x_test)

# Train Naïve Bayes classifier
classifier = GaussianNB()
classifier.fit(x_train, y_train)

# Predict on test set
y_pred = classifier.predict(x_test)

# Evaluation Metrics
accuracy = accuracy_score(y_test, y_pred)
precision = precision_score(y_test, y_pred)
recall = recall_score(y_test, y_pred)
cm = confusion_matrix(y_test, y_pred)

# Display results
print("Confusion Matrix:\n", cm)
print(f"Accuracy: {accuracy:.2f}")
print(f"Precision: {precision:.2f}")
print(f"Recall: {recall:.2f}")
